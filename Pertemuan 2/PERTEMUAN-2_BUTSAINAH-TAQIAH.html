Metode Hildreth-Lu

hildreth.lu.func<- function(r, model){
  x1 <- model.matrix(model)[,c(-1,-3)]
  x2 <- model.matrix(model)[,c(-1,-2)]
  y <- model.response(model.frame(model))
  n <- length(y)
  t <- 2:n
  y <- y[t]-r*y[t-1]
  x1 <- x1[t]-r*x1[t-1]
  x2 <- x2[t]-r*x2[t-1]
  
  return(lm(y~x1+x2))
}
